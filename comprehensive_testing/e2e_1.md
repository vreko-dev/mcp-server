```typescript
// src/test/suite/snapback-demo-flow.test.ts

import { expect, test } from "vscode-test-playwright";
import * as path from "path";
import * as fs from "fs";

/**
 * SnapBack Demo Flow Test
 *
 * This test demonstrates the core SnapBack workflow:
 * 1. Protect a TypeScript file
 * 2. Simulate AI-induced breaking change
 * 3. Trigger rollback
 * 4. Assert file is restored
 */

test.describe("SnapBack End-to-End Demo", () => {
  const testFilePath = "untitled:/snapback-demo.ts";
  const protectedCode = `
    export function greet(name: string): string {
      return \`Hello, \${name}!\`;
    }

    export function add(a: number, b: number): number {
      return a + b;
    }
  `;

  const brokenCode = `
    export function greet(name) {
      return 'Hello world';
    }

    // export function add removed by accident
  `;

  test("should protect, break, and restore code autonomously", async ({
    workbox,
    evaluateInVSCode,
    evaluateHandleInVSCode,
  }) => {
    // ==========================================
    // PHASE 1: Setup - Open editor with code
    // ==========================================

    const editorHandle = await evaluateHandleInVSCode(
      async (vscode, filePath) => {
        const document = await vscode.workspace.openTextDocument(
          vscode.Uri.parse(filePath)
        );
        const editor = await vscode.window.showTextDocument(document);
        return editor;
      },
      testFilePath
    );

    // Insert initial protected code
    await evaluateInVSCode(
      async (vscode, editor, code) => {
        await editor.edit((editBuilder) => {
          editBuilder.insert(new vscode.Position(0, 0), code);
        });
      },
      editorHandle,
      protectedCode
    );

    // ==========================================
    // PHASE 2: Protect the file with SnapBack
    // ==========================================

    // Trigger the "Protect with SnapBack" command
    // This command should be exposed by your extension
    await evaluateInVSCode(async (vscode) => {
      await vscode.commands.executeCommand("snapback.protect");
    });

    // Verify protection UI appears or state is set
    // Wait for a SnapBack indicator (badge, gutter icon, etc.)
    const protectedIndicator = workbox.locator(
      "[data-testid='snapback-protected-badge']"
    );

    await expect(protectedIndicator).toBeVisible({ timeout: 5000 });

    // Snapshot the protected state (internal checkpoint)
    const originalCheckpoint = await evaluateInVSCode(
      async (vscode) => {
        // Query your extension's state to confirm checkpoint created
        // e.g., get the protection manager's snapshot ID
        const snapbackModule = await vscode.extensions.getExtension(
          "snapback.snapback"
        );
        return snapbackModule?.exports?.getLastCheckpointId?.();
      }
    );

    expect(originalCheckpoint).toBeTruthy();

    // ==========================================
    // PHASE 3: Simulate AI-induced breaking change
    // ==========================================

    // Clear and replace with broken code (as AI might)
    await evaluateInVSCode(
      async (vscode, editor, broken) => {
        const fullRange = new vscode.Range(
          new vscode.Position(0, 0),
          editor.document.lineAt(editor.document.lineCount - 1).range.end
        );
        await editor.edit((editBuilder) => {
          editBuilder.replace(fullRange, broken);
        });
      },
      editorHandle,
      brokenCode
    );

    // Give the extension time to detect changes
    await new Promise((resolve) => setTimeout(resolve, 1000));

    // Verify the code is now broken (assertions would fail at type-check)
    const currentText = await evaluateInVSCode(
      async (vscode, editor) => {
        return editor.document.getText();
      },
      editorHandle
    );

    expect(currentText).toContain("export function greet(name)");
    expect(currentText).not.toContain("export function add");

    // ==========================================
    // PHASE 4: Trigger rollback
    // ==========================================

    // Access the rollback command or UI element
    // Option A: Via command palette
    await evaluateInVSCode(async (vscode) => {
      // This would be your custom command ID
      await vscode.commands.executeCommand("snapback.rollback");
    });

    // Option B: Via SnapBack status bar button (UI-driven)
    // const rollbackButton = workbox.getByRole('button', { name: /rollback/i });
    // await rollbackButton.click();

    // Wait for rollback to complete
    await new Promise((resolve) => setTimeout(resolve, 2000));

    // ==========================================
    // PHASE 5: Verify restoration
    // ==========================================

    const restoredText = await evaluateInVSCode(
      async (vscode, editor) => {
        return editor.document.getText();
      },
      editorHandle
    );

    // Assert original code is restored
    expect(restoredText).toContain("export function greet(name: string)");
    expect(restoredText).toContain("export function add(a: number, b: number)");
    expect(restoredText).not.toContain("export function greet(name) {");

    // Verify protection badge still shows (protection is still active)
    await expect(protectedIndicator).toBeVisible();

    // ==========================================
    // PHASE 6: Cleanup
    // ==========================================

    await editorHandle.release();
  });

  test("should show rollback history in webview", async ({
    workbox,
    evaluateInVSCode,
  }) => {
    // Open the SnapBack history panel via command
    await evaluateInVSCode(async (vscode) => {
      await vscode.commands.executeCommand("snapback.showHistory");
    });

    // Interact with webview to see checkpoints
    const historyPanel = workbox.locator("[data-testid='snapback-history']");
    await expect(historyPanel).toBeVisible({ timeout: 5000 });

    // Verify list of snapshots appears
    const snapshotItems = workbox.locator(
      "[data-testid='snapshot-item']"
    );
    await expect(snapshotItems.first()).toBeVisible();
  });

  test("should prevent protection on already-protected files", async ({
    evaluateInVSCode,
  }) => {
    const editorHandle = await evaluateHandleInVSCode(
      async (vscode, filePath) => {
        const document = await vscode.workspace.openTextDocument(
          vscode.Uri.parse(filePath)
        );
        return await vscode.window.showTextDocument(document);
      },
      testFilePath
    );

    // Protect once
    await evaluateInVSCode(async (vscode) => {
      await vscode.commands.executeCommand("snapback.protect");
    });

    // Try to protect again
    const result = await evaluateInVSCode(async (vscode) => {
      try {
        await vscode.commands.executeCommand("snapback.protect");
        return { success: true };
      } catch (err) {
        return { success: false, error: err.message };
      }
    });

    // Should either no-op gracefully or show a user message
    expect(result.success === false || result.success === true).toBe(true);

    await editorHandle.release();
  });
});
```

## Key Design Decisions

1. **Dual-layer testing**: Uses both `evaluateInVSCode()` for API calls (command execution, state checks) and `workbox` selectors for UI assertions (badges, panels, dialogs).

2. **Handle-based editor lifecycle**: Keeps a reference to the editor throughout the test so you can inspect and mutate it across phases. Explicit release prevents resource leaks.

3. **Explicit phase structure**: Each test is broken into named phases (Setup → Protect → Break → Rollback → Verify → Cleanup) so it reads like a user story and is easy to debug when something fails.

4. **Test data inline**: The protected and broken code are defined in the test, making intent crystal clear and avoiding file I/O brittleness.

5. **Headed + UI mode options**: You can run with `--headed` to *watch* the test drive VS Code, or `--ui` for Playwright Inspector to step through interactively.

6. **Trace capture**: `vscodeTrace: "on"` records full traces you can replay in Playwright Inspector to pinpoint selector or timing issues.

## Running it

```bash
# Run all tests headless
npm test

# Run only the demo flow with VS Code visible
npm run test:demo

# Run with Playwright Inspector (step through live)
npm test -- --debug
```

This structure scales. You can add more test cases for edge cases (multi-file protection, concurrent rollbacks, webview interactions) without redoing the orchestration.
