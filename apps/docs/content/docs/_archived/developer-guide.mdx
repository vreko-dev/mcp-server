## Package Architecture

### Monorepo Structure

SnapBack uses a monorepo with pnpm workspaces and Turborepo for build orchestration:

```
snapback/
├── apps/                      # Applications
│   ├── docs/                  # User documentation site
│   ├── web/                   # Main web application
│   ├── vscode/                # VS Code extension
│   ├── mcp-server/            # MCP server
│   ├── cli/                   # Command-line tool
│   └── api/                   # Backend API service ([Docs](/docs/api-reference))
├── packages/                  # Private packages
│   ├── contracts/             # Type definitions (public & private)
│   ├── core/                  # Core business logic
│   ├── infrastructure/        # Logging, metrics, tracing
│   ├── sdk/                   # SDK with full features
│   └── events/                # Event bus
├── packages-oss/              # Public packages
│   ├── contracts/             # Public type definitions
│   ├── infrastructure/        # Public utilities
│   ├── sdk/                   # Public SDK
│   ├── config/                # Configuration utilities
│   └── events/                # Event bus
└── pnpm-workspace.yaml        # Workspace config
```

### Package Naming

- **Private packages**: `@snapback/core`, `@snapback/sdk`, `@snapback/infrastructure`
- **Public packages**: `@snapback-oss/sdk`, `@snapback-oss/contracts`, `@snapback-oss/infrastructure`

## Import Boundaries

Proper import boundaries maintain architecture integrity and prevent circular dependencies.

### Dependency Graph

```
┌─────────────────────────────────┐
│           Apps Layer             │
│     (apps/vscode, apps/web)     │
│  CAN import: All packages        │
└──────────────┬──────────────────┘
               │
┌──────────────▼──────────────────┐
│    Business Logic Layer          │
│   (@snapback/core, /sdk)        │
│  CAN import: contracts, events, │
│              infrastructure      │
└──────────────┬──────────────────┘
               │
┌──────────────▼──────────────────┐
│      Foundation Layer            │
│  @snapback/contracts (types)    │
│  @snapback/events (pub/sub)     │
│  @snapback/infrastructure       │
│  CAN import: Only types         │
└─────────────────────────────────┘
```

### Import Rules

```typescript
// ✅ CORRECT - Use @snapback/* namespace for cross-package imports
import { logger } from "@snapback/infrastructure";
import type { Snapshot } from "@snapback/contracts";
import { SnapshotManager } from "@snapback/core";

// ❌ WRONG - Don't use relative paths across packages
import { logger } from "../../../packages/infrastructure/src/logging";

// ✅ CORRECT - Relative imports within the same package
import { validateSnapshot } from "./validation";
import type { SnapshotConfig } from "./types";

// ❌ WRONG - OSS packages importing private packages
// (in @snapback-oss/sdk)
import { analyticsClient } from "@snapback/infrastructure"; // ❌
```

### OSS Package Rules

Public packages (`@snapback-oss/*`) have stricter boundaries:

```typescript
// ✅ ALLOWED - Import from other OSS packages
import { logger } from "@snapback-oss/infrastructure";
import type { Snapshot } from "@snapback-oss/contracts";

// ❌ FORBIDDEN - NEVER import from private packages
import { AnalyticsManager } from "@snapback/core";          // ❌
import { posthog } from "@snapback/infrastructure/posthog"; // ❌
```

## Type Safety

SnapBack uses advanced TypeScript patterns for maximum type safety.

### Discriminated Unions

Use for state machines and result types:

```typescript
// ✅ State machine pattern
type SnapshotState =
  | { status: "loading" }
  | { status: "empty" }
  | { status: "error"; error: Error }
  | { status: "ready"; data: Snapshot };

// Type guards provide exhaustive checking
function handleState(state: SnapshotState): void {
  switch (state.status) {
    case "loading":
      showSpinner();
      break;
    case "empty":
      showEmptyState();
      break;
    case "error":
      showError(state.error); // TypeScript knows .error exists
      break;
    case "ready":
      showSnapshot(state.data); // TypeScript knows .data exists
      break;
  }
}
```

### Type Guards

Use `is` predicates for runtime type narrowing:

```typescript
// Define a type guard
function isSnapshot(value: unknown): value is Snapshot {
  return (
    typeof value === "object" &&
    value !== null &&
    "id" in value &&
    "content" in value &&
    typeof (value as any).id === "string"
  );
}

// Use it for type narrowing
const data: unknown = JSON.parse(jsonString);

if (isSnapshot(data)) {
  // TypeScript knows data is Snapshot here
  console.log(data.content);
} else {
  console.error("Invalid snapshot format");
}
```

### Const Assertions

Use for readonly data and enum alternatives:

```typescript
// ❌ Without const assertion - type is "string"
const protectionLevels = ["watch", "warn", "block"];

// ✅ With const assertion - types are readonly and specific
const PROTECTION_LEVELS = ["watch", "warn", "block"] as const;
type ProtectionLevel = typeof PROTECTION_LEVELS[number]; // "watch" | "warn" | "block"

// Type-safe validation function
function isProtectionLevel(value: string): value is ProtectionLevel {
  return (PROTECTION_LEVELS as readonly string[]).includes(value);
}
```

## Error Handling

Use the `Result<T, E>` pattern for error handling:

```typescript
// Error handling type
type Result<T, E = Error> =
  | { success: true; value: T }
  | { success: false; error: E };

// Constructor functions
function Ok<T>(value: T): Result<T> {
  return { success: true, value };
}

function Err<E>(error: E): Result<never, E> {
  return { success: false, error };
}

// Type guards
function isOk<T, E>(result: Result<T, E>): result is { success: true; value: T } {
  return result.success === true;
}

// Usage
async function createSnapshot(path: string): Promise<Result<Snapshot, Error>> {
  if (!path) {
    return Err(new Error("Invalid path"));
  }

  try {
    const snapshot = await storage.create(path);
    return Ok(snapshot);
  } catch (err) {
    return Err(err instanceof Error ? err : new Error(String(err)));
  }
}

// Handling results
const result = await createSnapshot("/path/to/file");

if (isOk(result)) {
  console.log("Created:", result.value.id);
} else {
  console.error("Failed:", result.error.message);
}
```

### When to Use Result vs Throwing

**Use `Result<T, E>` for:**
- Expected failures (missing file, validation errors)
- Recoverable errors (retry logic needed)
- Public APIs (SDKs, service boundaries)
- Composable operations

**Throw errors for:**
- Programming errors (null checks, type assertions)
- Initialization failures (database down)
- Invariant violations
- Development-time issues

## Testing

### Test Structure

```typescript
import { describe, it, expect, beforeEach, afterEach } from "vitest";
import { createSnapshot } from "@snapback/sdk";

describe("Snapshot Management", () => {
  let storage: MockStorage;

  beforeEach(() => {
    storage = new MockStorage();
  });

  afterEach(() => {
    storage.cleanup();
  });

  describe("createSnapshot", () => {
    it("should create a snapshot with valid path", async () => {
      const result = await createSnapshot("/valid/path.ts");

      expect(isOk(result)).toBe(true);
      if (isOk(result)) {
        expect(result.value).toHaveProperty("id");
        expect(result.value.filePath).toBe("/valid/path.ts");
      }
    });

    it("should return error for invalid path", async () => {
      const result = await createSnapshot("");

      expect(isErr(result)).toBe(true);
      if (isErr(result)) {
        expect(result.error.message).toContain("Invalid");
      }
    });

    it("should handle storage failures", async () => {
      storage.mockFailure = true;

      const result = await createSnapshot("/path.ts");

      expect(isErr(result)).toBe(true);
    });
  });
});
```

### Running Tests

```bash
# Run all tests
pnpm test

# Run tests for specific package
pnpm -F @snapback/sdk test

# Watch mode
pnpm test --watch

# With coverage
pnpm test --coverage

# UI mode
pnpm test --ui
```

## Performance

### Best Practices

1. **Lazy Load Dependencies**
   ```typescript
   // ✅ Import only what you need
   import { logger } from "@snapback/infrastructure";

   // ❌ Avoid barrel imports for large modules
   import * as infrastructure from "@snapback/infrastructure";
   ```

2. **Memoization**
   ```typescript
   // ✅ Cache expensive operations
   const cachedSnapshots = new Map<string, Snapshot>();

   function getSnapshot(id: string): Snapshot | undefined {
     return cachedSnapshots.get(id);
   }
   ```

3. **Batch Operations**
   ```typescript
   // ✅ Combine multiple operations
   async function createMultipleSnapshots(paths: string[]): Promise<Result<Snapshot[]>> {
     const results = await Promise.all(
       paths.map(path => createSnapshot(path))
     );

     const successful = results.filter(isOk).map(r => r.value);
     return Ok(successful);
   }
   ```
