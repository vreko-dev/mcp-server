```typescript
export default async function DocsLayoutWrapper({
  children,
}: {
  children: React.ReactNode;
}): Promise<ReactElement> {
  // Server-side host detection for subdomain routing
  const headersList = await headers();
  const host = headersList.get("host") || "";
  const protocol = headersList.get("x-forwarded-proto") || "http";

  // Dynamic home URL based on environment
  let homeUrl = "/";
  if (host.startsWith("docs.")) {
    const mainDomain = host.replace("docs.", "");
    homeUrl = `${protocol}://${mainDomain}`;
  } else if (host.includes("docs.localhost")) {
    const port = host.split(":")[1] || "3000";
    homeUrl = `${protocol}://localhost:${port}`;
  }

  return (
    <NextProvider>
      <RootProvider>
        <DocsLayout
          tree={source.pageTree}
          nav={{
            title: <HomeLinkTitle />,
            url: homeUrl,
            transparentMode: "top",
          }}
          links={[
            { text: "Home", url: homeUrl, icon: <HomeIcon /> },
            { text: "GitHub", url: "...", icon: <GitHubIcon />, external: true },
            { text: "Discord", url: "...", icon: <DiscordIcon />, external: true },
          ]}
          sidebar={{
            defaultOpenLevel: 1,
            banner: (
              <div>🧢 SnapBack Docs - Custom banner</div>
            ),
            collapsible: true,
          }}
        >
          {children}
        </DocsLayout>
      </RootProvider>
    </NextProvider>
  );
}
```

### Page (`app/docs/[[...slug]]/page.tsx`)

```typescript
export default async function DocsPageWrapper({
  params,
}: {
  params: Promise<{ slug?: string[] }>;
}) {
  const { slug = [] } = await params;
  const page = source.getPage(slug);

  if (!page) notFound();

  const MDX = (page.data as any).body;
  const components = useMDXComponents();

  return (
    <DocsPage toc={(page.data as any).toc}>
      <DocsBody>
        <MDX components={components} />
      </DocsBody>
    </DocsPage>
  );
}

export async function generateStaticParams() {
  return source.getPages().map((page: any) => ({
    slug: page.slugs,
  }));
}

export async function generateMetadata({
  params,
}: {
  params: Promise<{ slug?: string[] }>;
}): Promise<Metadata> {
  const { slug = [] } = await params;
  const page = source.getPage(slug);

  if (!page) notFound();

  return {
    title: page.data.title,
    description: page.data.description,
  };
}
```

**Features**:
- Catch-all route with `[[...slug]]`
- Static generation with `generateStaticParams`
- Automatic metadata from frontmatter
- Custom MDX components injection
- TOC (Table of Contents) support

## 10. Blog Implementation (Separate System)

**Location**: `apps/web/content/posts/`

**Loader**: `modules/marketing/blog/utils/lib/posts.ts`

```typescript
const contentDirectory = path.join(process.cwd(), "content/posts");

function loadAllPosts(): Post[] {
  const fileNames = fs.readdirSync(contentDirectory);

  const allPosts = fileNames
    .filter((fileName) => fileName.endsWith(".md") || fileName.endsWith(".mdx"))
    .map((fileName) => {
      // Extract locale from filename (e.g., "first-post.de.mdx" -> "de")
      const match = fileName.match(/^(.+?)(?:\.([a-z]{2}))?\.mdx?$/);
      const [, baseName, locale] = match;

      const fullPath = path.join(contentDirectory, fileName);
      const fileContents = fs.readFileSync(fullPath, "utf8");
      const { data, content } = matter(fileContents);

      return {
        path: baseName,
        locale: locale || "en",
        title: data.title,
        date: data.date,
        image: data.image,
        authorName: data.authorName,
        authorImage: data.authorImage,
        excerpt: data.excerpt,
        tags: data.tags,
        published: data.published ?? false,
        body: content,
      };
    })
    .filter((post) => post.published);

  return allPosts;
}
```

**Key Differences from Docs**:
- Uses `gray-matter` for frontmatter parsing
- File system-based (not Fumadocs)
- Locale support via filename pattern (`post.de.mdx`)
- `published` flag for draft control
- More extensive frontmatter (author, image, tags, etc.)
